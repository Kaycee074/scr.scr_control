83  /**
84  * Detaches the source from connected sinks
85  */
86  void detachAll(void) {
87  iris::synchronization::Lock lock(mutex);
88  // detach all connected sinks
89  while (!pictureSinks.empty()) {
90  typename SinkTypeList::iterator i = pictureSinks.begin();
91  SinkType* sink = *i;

93  }

112  /**

115  * @param data Data to be propagated
116  * @param index 0-based index of the data if the total number of
117  * data objects is known, 0 otherwise
118  * @param indexCount Total number of data objects (0 if unknown)
119  */
120  void pushDataToSinks(T data, unsigned long index, unsigned long indexCount) {
121  iris::synchronization::Lock lock(mutex);
122  // pass on data to all sinks' processNewPicture() method
123  for (typename SinkTypeList::const_iterator i = pictureSinks.begin(); i
124  != pictureSinks.end(); ++i) {
125  SinkType* sink = *i;
126  sink->processNewData(data, index, indexCount);
127  }
128  }
129 
130  void detachSink(SinkType* sink, bool detachCall) {
131  iris::synchronization::Lock lock(mutex);
132  typename SinkTypeList::iterator it = std::find(pictureSinks.begin(), pictureSinks.end(), sink);
133  if (sink && it != pictureSinks.end()) {
134  pictureSinks.erase(it);
135  if (detachCall) {
136  sink->detach();
137  }
138  }
139  }
140 
141  void attachSink(SinkType* sink, bool attachCall) {
142  iris::synchronization::Lock lock(mutex);
143  typename SinkTypeList::iterator it = std::find(pictureSinks.begin(), pictureSinks.end(), sink);
144  if (sink && it == pictureSinks.end()) {
145  pictureSinks.push_back(sink);
146  if (attachCall) {
147  sink->attachSource(this);
148  }
149  }
150  }
